allow read, write: if request.auth.token.role == "super_admin"; allow read: if request.auth.token.tenantId == tenantId; allow write: if request.auth.token.tenantId == tenantId && request.auth.token.role in ["tenant_admin","rt_admin","bendahara","security"];
